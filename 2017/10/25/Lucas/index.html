    Lucas定理

    Lucas定理是用来求C（n,m）mod p 且p为质数时的方法。在noip范围内这基本是求该类问题最优的方法，不需考虑到适用范围的问题。

    ++
    namespace Lucas_Method
    {
    typedef long long ll;

    ll quick_pow(ll a , ll b ,ll p) {
    ll ans = 1;
    for(; b ; b >>= 1 , a = a*a%p)
        if(b & 1) ans = ans * a % p;
    return ans;
}

ll Calc(ll a , ll b ,ll p) {
    if(a < b) return 0;
    else if(a == b) return 1;

    if(b > a-b) b = a-b;
    ll ans = 1 , ca = 1 , cb = 1;
    for(ll i = 0 ; i < b ; i++) {
        ca = (ca * (a-i)) % p;
        cb = (cb * (b-i)) % p;
    }
    ans = (ca*quick_pow(cb,p-2,p))%p;
    return ans;
}
ll ans = 1;
ll Lucas(int n , int m , ll p) {
    ans = (ans * Calc(n%p , m%p ,p)) %p;
    if(!n || !m || !ans) return ans;
    return Lucas(n/p , m/p , p);
}
